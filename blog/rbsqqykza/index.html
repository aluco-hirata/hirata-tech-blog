グローバルに定義するとどこからでも参照できるため、関数スコープ内に変数を定義

    console.log(num);

・クロージャー使用

function incrementFactory() {
    let num = 0

    function increment() {
        num = num + 1
        console.log(num);
    }
    return increment
}

const increment = incrementFactory()

ケース02・・・動的関数

function addNumberFactory(num) {
    function addNumber(value) {
        return num + value;
    }
    return addNumber;
}

const add5 = addNumberFactory(5);
const result = add5(5);
const result = add5(10);
